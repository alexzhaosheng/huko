root@huko:~$ ./launch

huko-explicit

Minimalist CLI. Maximalist intelligence. Configurable all the way down.

[01] Noun-first CLI

# continue the active session
huko run "review this diff for risky changes"
# check who the agent is talking to
huko sessions current
# set the default model explicitly
huko model default anthropic/claude-sonnet-4-6

[02] Explicit Setup

# start with the guided path
huko setup
# add a provider definition
huko provider add --name=OpenRouter --protocol=openai --base-url=https://openrouter.ai/api/v1 --api-key-ref=openrouter
# bind the key to a named ref
huko keys set openrouter sk-or-...

[03] Layered Config

# inspect the resolved runtime config
huko config show
# inspect the merged infra view
huko info
# inspect only the global layer
huko info global

[04] Project Memory

# continue work inside this repo
huko run "continue the refactor plan"
# list saved sessions for this project
huko sessions list
# branch into a fresh thread
huko run --new --title="release triage" "summarize today's blockers"

[05] Zero-trace Runs

# run without touching session storage
huko run --memory "summarize this build log in 3 bullets"
# return machine-readable output
huko run --memory --json "return a release note draft"
# ask something private and leave no trail
huko run --memory "private question"

[06] Inspectable Agents

# capture a final result as JSON
huko run --json "generate a changelog from these commits"
# stream semantic events line by line
huko run --jsonl "fetch the page and extract the title"
# inspect the project-specific agent surface
huko info project
